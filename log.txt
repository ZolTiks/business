# Формула расчета финального дохода с бизнеса.
#
# Плейсхолдеры используемые по умолничаю:
# %business_income% # Базовый доход расчитываемый конфигурационной формулой индивидуально для каждого бизнеса.
# %player_business_income_multiplier% # Индивидуальный множитель дохода от любого бизнеса присвоенный конкретному игроку.
# %total_booster_business_income_multiplier% # Суммарный "x2 + x2 = x3, а не x4" множитель всех бустеров игрока если есть или 1.
# %total_cloth_business_income_multiplier% # Суммарный "x2 + x2 = x3, а не x4" множитель всех одетых одежд игрока если есть или 1.
#
# Плейсхолдеры PAPI поддерживаются тоже.
business_final_income_formula: '(%business_income% + %business_income% * %player_business_income_multiplier%) * %total_booster_business_income_multiplier% * %total_cloth_business_income_multiplier%'
